    // Add event listener for delete button
    document
      .getElementById(`suprimer_${index}`)
      .addEventListener("click", () => {
        movies.splice(index, 1);
        localStorage.setItem("movies", JSON.stringify(movies));
        renderMovies(); 
      });

    // Add event listener for update button
    document
      .getElementById(`mise_a_jour_${index}`)
      .addEventListener("click", () => {
       
        input_title.value = movie.title;
        input_realisateur.value = movie.realisateur;
        input_genre.value = movie.genre;
        input_url.value = movie.imageurl;
        input_date.value = movie.date;

      });

  // Modify the "ajouter" button to act as an update button
  ajouter.textContent = "Update Movie";
  ajouter.onclick = function () {
    movies[index] = {
      title: input_title.value,
      realisateur: input_realisateur.value,
      genre: input_genre.value,
      imageurl: input_url.value,
      date: input_date.value,
    };

    localStorage.setItem("movies", JSON.stringify(movies));
    displayMovies();
    resetForm();
  };
